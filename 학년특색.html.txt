<script>
    // 네가 직접 산출해서 제공해 준 4대 분야별 가중치 정밀 벡터 데이터 (w1 ~ w8)
    const fieldWeights = {
        '재료공학분야': [10, 10, 20, 10, 10, 10, 10, 20],
        '항공우주분야': [25, 22, 20, 12, 11, 4,  3,  3],
        '건축 분야':   [20, 18, 16, 15, 13, 8,  7,  3],
        '생명분야':     [10, 10, 20, 10, 10, 10, 10, 20] // 재료공학분야와 동일한 규칙 반영
    };

    // 물성별 데이터 분석 경계 조건 상수값 설정
    const bounds = {
        x1: { min: 0.01, max: 10, isReverse: true },
        x2: { min: 85, max: 3245, isReverse: false },
        x3: { min: 0.5, max: 130000, isReverse: false },
        x4: { min: 500, max: 6150, isReverse: true },
        x5: { min: 1, max: 4000, isReverse: false },
        x6: { min: 0, max: 90, isReverse: false },
        x7: { min: 0.2, max: 4000, isReverse: false },
        x8: { min: 0, max: 100000, isReverse: false }
    };

    let fieldChart = null;
    let criteriaChart = null;

    function updateAnalysis() {
        const matName = document.getElementById('material-name').value || '이름 없음';
        document.getElementById('chart-title').innerText = `📊 '${matName}'의 분야별 최종 적합도 결과`;

        const rawValues = {
            x1: parseFloat(document.getElementById('score-x1').value),
            x2: parseFloat(document.getElementById('score-x2').value),
            x3: parseFloat(document.getElementById('score-x3').value),
            x4: parseFloat(document.getElementById('score-x4').value),
            x5: parseFloat(document.getElementById('score-x5').value),
            x6: parseFloat(document.getElementById('score-x6').value),
            x7: parseFloat(document.getElementById('score-x7').value),
            x8: parseFloat(document.getElementById('score-x8').value)
        };

        // 실시간 수치 텍스트 표기 동기화
        Object.keys(rawValues).forEach(key => {
            document.getElementById(`val-${key}`).innerText = rawValues[key];
        });

        // 1단계: 최대-최소 정규화 수식 변환 (X -> X')
        const normalizedXPrime = [];
        Object.keys(bounds).forEach(key => {
            const b = bounds[key];
            const x = rawValues[key];
            let xPrime = 0;
            
            if (b.isReverse) {
                // 역방향 정규화 적용 (내화학성, 경량성 밀도용)
                xPrime = (b.max - x) / (b.max - b.min);
            } else {
                // 일반 정규화 적용
                xPrime = (x - b.min) / (b.max - b.min);
            }
            normalizedXPrime.push(xPrime);
        });

        // 2단계: 분야별 적합도 선형 결합 연산 (W * X')
        const labels = Object.keys(fieldWeights);
        const suitabilityScores = [];

        labels.forEach(field => {
            const weights = fieldWeights[field];
            let dotProduct = 0;
            
            for (let i = 0; i < 8; i++) {
                dotProduct += weights[i] * normalizedXPrime[i];
            }
            suitabilityScores.push(dotProduct.toFixed(2));
        });

        // 우측: 100점 만점 기준 분야별 적합도 막대 그래프 렌더링
        const ctxField = document.getElementById('fieldChart').getContext('2d');
        if (fieldChart) fieldChart.destroy();
        
        fieldChart = new Chart(ctxField, {
            type: 'bar',
            data: {
                labels: labels,
                datasets: [{
                    label: `${matName}의 종합 분야 적합도 (% 단위 / 100점 만점)`,
                    data: suitabilityScores,
                    backgroundColor: ['rgba(59, 130, 246, 0.7)', 'rgba(244, 63, 94, 0.7)', 'rgba(245, 158, 11, 0.7)', 'rgba(16, 185, 129, 0.7)'],
                    borderColor: ['rgba(59, 130, 246, 1)', 'rgba(244, 63, 94, 1)', 'rgba(245, 158, 11, 1)', 'rgba(16, 185, 129, 1)'],
                    borderWidth: 1.5
                }]
            },
            options: {
                responsive: true,
                scales: { y: { beginAtZero: true, max: 100 } }
            }
        });
    }

    // 하단: 네가 고안한 8차원 가중치 레이더 차트 생성
    function drawCriteriaChart() {
        const ctxCriteria = document.getElementById('criteriaChart').getContext('2d');
        const fields = Object.keys(fieldWeights);
        const datasets = [];
        const colors = [
            { bg: 'rgba(59, 130, 246, 0.05)', border: 'rgba(59, 130, 246, 1)' },  
            { bg: 'rgba(244, 63, 94, 0.05)', border: 'rgba(244, 63, 94, 1)' },  
            { bg: 'rgba(245, 158, 11, 0.05)', border: 'rgba(245, 158, 11, 1)' }, 
            { bg: 'rgba(16, 185, 129, 0.05)', border: 'rgba(16, 185, 129, 1)' }  
        ];

        fields.forEach((field, index) => {
            datasets.push({
                label: field,
                data: fieldWeights[field],
                backgroundColor: colors[index].bg,
                borderColor: colors[index].border,
                borderWidth: 1.5,
                pointRadius: 2
            });
        });

        criteriaChart = new Chart(ctxCriteria, {
            type: 'radar',
            data: {
                labels: ['내화학성', '내열성', '기계적강도', '경량성', '표면적', '다공성', '열전도성', '전기전도성'],
                datasets: datasets
            },
            options: {
                responsive: true,
                scales: { r: { suggestedMin: 0, suggestedMax: 30 } }
            }
        });
    }

    window.onload = function() {
        updateAnalysis();
        drawCriteriaChart();
    };
</script>
